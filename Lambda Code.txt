import json
import boto3

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('Accounts')

def lambda_handler(event, context):

    accountId = event['queryStringParameters']['accountId']

    response = table.get_item(
        Key={
            'accountId': accountId
        }
    )

    return {
        'statusCode': 200,
        'body': json.dumps(response['Item'])
    }
